Guessing Game in C

This is a C program is game that will ask you to guess a number.

Code:


#include

int main(){
   
  int rn, guess, t, tries, j, i, n, m; 
  char again, d;
  char name[50];
  
    srand(time(0));
   
   printf("what's your name?: ");
   scanf("%s", &name);
   
  while(d != 'e' && d != 'E' && d != 'm' && d != 'M' && d != 'h' && d != 'H'){
   printf("difficulty level: (E/M/H): ");
   scanf(" %c", &d);
   

   if(d == 'e' || d =='E'){
       m = 50;
       n = 1;
   }else if(d == 'm' || d == 'M'){
       m = 100;
       n = 1;
   }
   else if(d == 'h' || d == 'H'){
       m = 1000;
       n = 1;
   }else{
       printf("invalid try again\n");
   }
  }
   
   
   do{ 
    rn = rand() % m + n;
    guess = 0;
    t = 0;
    tries = 10;
    j = 2;
    i = j*j;
    
    
    printf("\nwelcome to number guessing game!\n");
    printf("think a number between %d-%d\n", n, m);
    
    while(guess != rn && tries > 0){
        t++;
        tries--;
        printf("enter a number: ");
        scanf("%d", &guess);
        if(guess > rn){
            printf("\ntoo high, try again       tries left:%d\n", tries);
        }
        else if(guess < rn){
            printf("\ntoo low, try again        tries left:%d\n", tries);
        }
        else if(guess == rn){
            printf("\ncongrats you got it in %d tries\n", t);
            break;
        }
        if(tries == 0){
            printf("\nyou're out of tries, the number is %d\n", rn);
            continue;
        }
        }
        if(rn % i == 0){
            printf("random number is not prime\n");
        }
        else{
            printf("random number is prime!\n");
        }
    
   
    printf("\ndo you want to play again?(y/n): ");
    scanf(" %c", &again);
   }while(again == 'y' || again == 'Y');
   
   printf("\nbye");
    
    return 0;
}

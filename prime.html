Prime Numbers Finders in C

This is a C program that finds prime numbers between two numbers entered by the user.

Code:


#include 

int main(){
    
    int first, last, i, j, prime, count;
    float temp;
    
    do{
        printf("enter the first number: ");
        if(scanf("%f", &temp) != 1 || temp != (int)temp || temp < 0){
            printf("invalid, enter positivie whole number\n\n");
            while(getchar() != '\n');
            continue;
        }
        
        first = (int)temp;
        
        
        printf("enter the last number: ");
        if(scanf("%f", &temp) != 1 || temp != (int)temp || temp < 0 || temp > 1000){
            printf("invalid, enter positivie whole number\n\n");
            while(getchar() != '\n');
            continue;
        }
        
        last = (int)temp;
        
        if(first > last){
            printf("make sure first number is less than or equal to last number\n");
        }
        
    }while(first < 0 || last < 0 || first > last);
    
    printf("prime number are:\n");
    
    count = 0;
    for(i = first; i <= last; i++){
        if(i < 2)
        continue;
        
        prime = 1;
        
        for(j = 2; j * j <= i; j++){
        if(i % j == 0){
            prime = 0;
            break;
        }
        } 
        if(prime){
            printf("%d ", i);
            count++;
        }
        
    }
    
    if(count == 0)
        printf("no prime number found");
    
    else printf("count:%d", count);
    
return 0;
}

You can run this code in any C compiler to see the results.
